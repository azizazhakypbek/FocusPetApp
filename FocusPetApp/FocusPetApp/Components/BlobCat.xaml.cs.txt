using System.Windows;
using System.Windows.Controls;
using System.Windows.Media;
using System.Windows.Media.Animation;

namespace FocusPetApp.Components
{
    public partial class BlobCat : UserControl
    {
        private Storyboard _blinkSb;
        private Storyboard _breathSb;

        public BlobCat()
        {
            InitializeComponent();
            _blinkSb = (Storyboard)Resources["BlinkSb"];
            _breathSb = (Storyboard)Resources["BreathSb"];
            _breathSb.Begin(this, true);
        }

        // --- Public API for animations ---

        public void Blink()
        {
            _blinkSb.Begin(this, true);
        }

        public void Idle()
        {
            FaceShift.X = 0; FaceShift.Y = 0;
            LidLeft.Opacity = 0; LidRight.Opacity = 0;
        }

        public void LookLeft()  { FaceShift.X = -5; }
        public void LookRight() { FaceShift.X =  5; }
        public void LookUp()    { FaceShift.Y = -4; }
        public void LookDown()  { FaceShift.Y =  4; }

        public void Squint()
        {
            LidLeft.Opacity = 0.7;
            LidRight.Opacity = 0.7;
        }

        // --- Customization ---

        public void SetColor(string theme)
        {
            // Body + ears share same colors
            (Color fill, Color stroke) = theme switch
            {
                "Pink"  => ( (Color)ColorConverter.ConvertFromString("#FAD4E5"),
                             (Color)ColorConverter.ConvertFromString("#C27C88") ),
                "Beige" => ( (Color)ColorConverter.ConvertFromString("#F5E6C8"),
                             (Color)ColorConverter.ConvertFromString("#C1A574") ),
                _       => ( (Color)ColorConverter.ConvertFromString("#B8F2D8"),
                             (Color)ColorConverter.ConvertFromString("#66A6C1") ), // Mint default
            };

            var bodyBrush = new SolidColorBrush(fill);
            var strokeBrush = new SolidColorBrush(stroke);
            Body.Fill = bodyBrush; Body.Stroke = strokeBrush;
            EarLeft.Fill = bodyBrush; EarLeft.Stroke = strokeBrush;
            EarRight.Fill = bodyBrush; EarRight.Stroke = strokeBrush;

            // Eyelids match body fill
            LidLeft.Fill = bodyBrush;
            LidRight.Fill = bodyBrush;
        }

        public void SetHat(bool on)
        {
            Hat.Visibility = on ? Visibility.Visible : Visibility.Collapsed;
        }
    }
}
