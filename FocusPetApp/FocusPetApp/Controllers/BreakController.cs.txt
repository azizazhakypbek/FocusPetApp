using System;

namespace FocusPetApp.Controllers
{
    public class BreakController
    {
        public int BreaksRemaining { get; private set; }

        public BreakController(int count)
        {
            BreaksRemaining = count;
        }

        public bool UseBreak()
        {
            if (BreaksRemaining <= 0) return false;
            BreaksRemaining--;
            return true;
        }
    }
}
